{
  "foo": "abcdefghijklmnopqrstuvwxyz",
  "test_var": "foo"
}
---
valid strings
:-2 is {{ foo[:-2] }}
:5 is {{ foo[:5] }}
2: is {{ foo[2:] }}
-3: is {{ foo[-3:] }}
3:5 is {{ foo[3:5] }}

valid lists
:1 is {{ [1,2,3][:1] }}
1:2 is {{ [1,2,3][1:2] }}
-2: is {{ [1,2,3][:-2] }}
-2: is {{ [1,2,3][2:] }}
-2: is {{ [1,2,3][-1:-2] }}
-1:3 is {{ [1,2,3][-1:3] }}
-1:-2 is {{ [1,2,3][-1:-2] }}

slicing variables
{{ foo[(test_var|length):] }}
{{ foo[1:(test_var|length)] }}
{{ foo[(test_var|length):-1] }}
{{ foo[(test_var|length):5] }}
